function myFunction(id) {

        x.className += " w3-show";
        x.previousElementSibling.className += " w3-theme-d1";

        x.className = x.className.replace("w3-show", "");
        x.previousElementSibling.className = 
        x.previousElementSibling.className.replace(" w3-theme-d1", "");
    }
}
